\noindent 效果展示:

\begin{algorithm}[H]
\caption{最大连续子数组和 MaxSubArray($X, low, high$)}
\KwIn{数组 $X$, 数组下标 $low, high$}
\KwOut{最大连续子数组和 $S_{\max}$}
\If{$low = high$}{
    \Return $X[low]$
}
\Else{
    $mid \leftarrow \lfloor \frac{low + high}{2} \rfloor$\;
    $S_1 \leftarrow$ MaxSubArray($X, low, mid$)\;
    $S_2 \leftarrow$ MaxSubArray($X, mid + 1, high$)\;
    $S_3 \leftarrow$ CrossingSubArray($X, mid, low, high$)\;
    $S_{\max} \leftarrow \max\{S_1, S_2, S_3\}$\;
    \Return{$S_{\max}$}
}
\end{algorithm}

\noindent 注意: 需要 \verb|$$| 才能显示数学字符
